// Main Dashboard Application import { SynapseRuntime } from '@snps/core'; import { HttpClient } from '@snps/http-client-rust'; class SynapseDashboard { private runtime: SynapseRuntime; private httpClient: HttpClient; constructor() { this.runtime = new SynapseRuntime(); this.httpClient = new HttpClient(); } public async initialize(): Promise<void> { await this.runtime.start(); await this.loadInitialData(); } private async loadInitialData(): Promise<void> { const metrics = await this.loadMetrics(); this.state.setState('metrics', metrics); } }
// Rust HTTP Client Implementation use reqwest; use serde_json; pub struct HttpClient { client: reqwest::Client, } impl HttpClient { pub fn new() -> Self { Self { client: reqwest::Client::new(), } } pub async fn get(&self, url: &str) -> Result<String, reqwest::Error> { let response = self.client.get(url).send().await?; let body = response.text().await?; Ok(body) } }
